import Foundation
import CoreMedia
import Observation

/// View model for timeline UI, bridging views to data layer and coordinators.
/// @Observable for SwiftUI reactive updates
@MainActor
@Observable
final class TimelineViewModel {
    /// Reference to the editor state
    private(set) var editorState: EditorState

    /// Layout cache coordinator
    private(set) var layoutCache: ClipLayoutCache

    /// Snapping coordinator
    private(set) var snappingCoordinator: SnappingCoordinator

    /// Undo/redo coordinator
    private(set) var undoRedoCoordinator: UndoRedoCoordinator

    /// Currently selected clip IDs
    private(set) var selectedClipIDs: Set<UUID> = []

    /// Current playhead position
    private(set) var playheadPosition: CMTime

    /// Current drag state
    private(set) var dragState: TimelineDragState = .idle

    /// Clip being dragged
    private(set) var draggedClip: VideoClip?

    /// Drag offset from original position
    private(set) var dragOffset: CGSize = .zero

    /// Snap indicator for visual feedback
    private(set) var snapIndicator: SnapIndicator?

    /// Visible time range (for virtualization)
    private(set) var visibleTimeRange: CMTimeRange

    /// Creates a new timeline view model
    /// - Parameter editorState: The editor state
    init(editorState: EditorState) {
        self.editorState = editorState
        self.layoutCache = ClipLayoutCache()
        self.snappingCoordinator = SnappingCoordinator()
        self.undoRedoCoordinator = UndoRedoCoordinator(editorState: editorState)
        self.playheadPosition = .zero
        self.visibleTimeRange = CMTimeRange(start: .zero, duration: CMTime(seconds: 60, preferredTimescale: 600))
    }
}

/// Visual snap indicator for drag operations
struct SnapIndicator {
    /// Position of the snap line
    let position: CGFloat
    /// Type of snap
    let snapType: SnapPointType
    /// Description of snap target
    let description: String
}
```

extension TimelineViewModel {
    /// Selects a single clip
    /// - Parameter clipID: ID of clip to select
    func selectClip(_ clipID: UUID) {
        selectedClipIDs = [clipID]
        postSelectionChangeNotification()
    }

    /// Selects multiple clips
    /// - Parameter clipIDs: IDs of clips to select
    func selectClips(_ clipIDs: [UUID]) {
        selectedClipIDs = Set(clipIDs)
        postSelectionChangeNotification()
    }

    /// Deselects all clips
    func deselectAll() {
        selectedClipIDs = []
        postSelectionChangeNotification()
    }

    /// Checks if a clip is selected
    /// - Parameter clipID: ID of clip to check
    /// - Returns: True if clip is selected
    func isClipSelected(_ clipID: UUID) -> Bool {
        return selectedClipIDs.contains(clipID)
    }

    private func postSelectionChangeNotification() {
        NotificationCenter.default.post(
            name: .timelineClipSelectionChanged,
            object: self,
            userInfo: ["selectedClipIDs": selectedClipIDs]
        )
    }
}

import SwiftUI
import CoreMedia

struct TimeRulerView: View {
    /// Visible time range to display
    let visibleTimeRange: CMTimeRange

    /// Current playhead position
    let playheadPosition: CMTime

    /// Time scale (pixels per second)
    let timeScale: CGFloat

    /// Height of the ruler
    let height: CGFloat = 30

    var body: some View {
        Canvas { context, size in
            // Draw background
            context.fill(
                Path(CGRect(origin: .zero, size: size)),
                with: .color(Color(nsColor: .systemGray6))
            )

            // Draw time markers
            drawTimeMarkers(in: context, size: size)

            // Draw playhead
            drawPlayhead(in: context, size: size)
        }
    }

    private func drawTimeMarkers(in context: GraphicsContext, size: CGSize) {
        let startSeconds = CMTimeGetSeconds(visibleTimeRange.start)
        let durationSeconds = CMTimeGetSeconds(visibleTimeRange.duration)

        // Determine marker interval based on zoom level
        let interval: TimeInterval
        if durationSeconds < 10 {
            interval = 1.0 // Every second
        } else if durationSeconds < 60 {
            interval = 5.0 // Every 5 seconds
        } else {
            interval = 10.0 // Every 10 seconds
        }

        // Draw markers
        var currentTime = ceil(startSeconds / interval) * interval
        while currentTime <= startSeconds + durationSeconds {
            let x = (currentTime - startSeconds) * timeScale

            // Draw tick mark
            let isMajorMarker = currentTime.truncatingRemainder(dividingBy: interval * 5) == 0
            let tickHeight: CGFloat = isMajorMarker ? 12 : 6

            context.stroke(
                Path { path in
                    path.move(to: CGPoint(x: x, y: size.height - tickHeight))
                    path.addLine(to: CGPoint(x: x, y: size.height))
                },
                with: .color(Color(nsColor: .labelColor)),
                lineWidth: 1
            )

            // Draw time label for major markers
            if isMajorMarker {
                let text = Text("\(Int(currentTime))s")
                    .font(.system(size: 10))
                    .foregroundColor(Color(nsColor: .secondaryLabelColor))

                context.draw(text, at: CGPoint(x: x + 4, y: size.height - 20))
            }

            currentTime += interval
        }
    }

    private func drawPlayhead(in context: GraphicsContext, size: CGSize) {
        let playheadX = (CMTimeGetSeconds(playheadPosition) - CMTimeGetSeconds(visibleTimeRange.start)) * timeScale

        // Draw playhead line
        context.stroke(
            Path { path in
                path.move(to: CGPoint(x: playheadX, y: 0))
                path.addLine(to: CGPoint(x: playheadX, y: size.height))
            },
            with: .color(.red),
            lineWidth: 2
        )

        // Draw playhead triangle
        let trianglePath = Path { path in
            path.move(to: CGPoint(x: playheadX - 6, y: 0))
            path.addLine(to: CGPoint(x: playheadX + 6, y: 0))
            path.addLine(to: CGPoint(x: playheadX, y: 8))
            path.closeSubpath()
        }

        context.fill(trianglePath, with: .color(.red))
    }
}
```

import SwiftUI
import CoreMedia

struct ClipView: View {
    /// The clip to display
    let clip: VideoClip

    /// Calculated frame for the clip
    let frame: CGRect

    /// Whether the clip is selected
    let isSelected: Bool

    /// Whether the clip is being dragged
    let isDragging: Bool

    /// Action when clip is pressed
    let onPress: () -> Void

    /// Action when clip is dragged
    let onDrag: (CGSize) -> Void

    var body: some View {
        ZStack {
            // Clip background
            Rectangle()
                .fill(Color(nsColor: .controlAccentColor).opacity(isDragging ? 0.8 : 0.6))
                .frame(width: frame.width, height: frame.height)

            // Clip name
            Text(clip.name)
                .font(.system(size: 11))
                .foregroundColor(.white)
                .padding(4)
                .frame(maxWidth: .infinity, alignment: .leading)
        }
        .overlay(
            // Selection indicator
            Group {
                if isSelected {
                    Rectangle()
                        .stroke(Color.blue, lineWidth: 3)
                } else if isDragging {
                    Rectangle()
                        .stroke(Color.gray.opacity(0.5), lineWidth: 2)
                }
            }
        )
        .gesture(
            DragGesture(minimumDistance: 0)
                .onChanged { value in
                    if dragState == .idle {
                        onPress()
                    }
                    onDrag(value.translation)
                }
        )
        .shadow(radius: isDragging ? 8 : 2)
    }

    @Observable private class DragStateHolder {
        @Observation ignored var dragState: TimelineDragState = .idle
    }

    @State private var dragStateHolder = DragStateHolder()
    private var dragState: TimelineDragState {
        get { dragStateHolder.dragState }
        set { dragStateHolder.dragState = newValue }
    }
}

import SwiftUI

struct ClipTrackView: View {
    /// The track to display
    let track: ClipTrack

    /// Layout cache for clip positions
    @Binding var layoutCache: ClipLayoutCache

    /// Selected clip IDs
    let selectedClipIDs: Set<UUID>

    /// Action when clip is selected
    let onClipSelected: (UUID) -> Void

    /// Action when clip is dragged
    let onClipDragged: (UUID, CGSize) -> Void

    var body: some View {
        VStack(spacing: 0) {
            // Track header
            HStack {
                Text(track.name)
                    .font(.system(size: 12, weight: .medium))
                    .foregroundColor(.primary)

                Spacer()

                if !track.isEnabled {
                    Text("Muted")
                        .font(.system(size: 10))
                        .foregroundColor(.secondary)
                }
            }
            .padding(.horizontal, 8)
            .padding(.vertical, 4)
            .background(Color(nsColor: .controlBackgroundColor))

            // Clips
            ZStack(alignment: .leading) {
                // Track background
                Rectangle()
                    .fill(Color(nsColor: .systemGray6))
                    .frame(height: 60)

                // Clips
                ForEach(track.clips) { clip in
                    let frame = layoutCache.updateLayout(for: clip, in: track)

                    ClipView(
                        clip: clip,
                        frame: frame,
                        isSelected: selectedClipIDs.contains(clip.id),
                        isDragging: false
                    ) {
                        onClipSelected(clip.id)
                    } onDrag: { offset in
                        onClipDragged(clip.id, offset)
                    }
                    .offset(x: frame.x, y: frame.y)
                    .frame(width: frame.width, height: frame.height)
                }
            }
            .frame(height: 60)
        }
    }
}

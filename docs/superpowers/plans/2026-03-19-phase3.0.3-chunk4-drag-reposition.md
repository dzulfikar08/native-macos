```swift
extension TimelineViewModel {
    /// Starts a drag operation for a clip
    /// - Parameters:
    ///   - clip: The clip being dragged
    ///   - position: The starting position in view coordinates
    func startDrag(clip: VideoClip, at position: CGPoint) {
        guard dragState.canStartDrag else {
            return
        }

        draggedClip = clip
        originalTrackID = clip.trackID
        dragStartPosition = position
        dragOffset = .zero

        // Calculate original frame
        if let track = tracks.first(where: { $0.clips.contains(where: { $0.id == clip.id }) }) {
            originalClipFrame = layoutCache.updateLayout(for: clip, in: track)
        }

        dragState = .dragStarted

        // Post notification
        NotificationCenter.default.post(
            name: .timelineClipDragging,
            object: self,
            userInfo: ["clipID": clip.id, "position": position]
        )
    }
}
```

- [ ] **Step 3: Implement updateDrag method**

```swift
extension TimelineViewModel {
    /// Updates drag position with snap calculation
    /// - Parameter position: Current drag position in view coordinates
    func updateDrag(at position: CGPoint) {
        guard dragState.isDragging, let clip = draggedClip else {
            return
        }

        dragState = .dragging

        // Calculate offset from start
        let offset = CGSize(
            width: position.x - dragStartPosition.x,
            height: position.y - dragStartPosition.y
        )
        dragOffset = offset

        // Convert x offset to time offset
        let timeOffset = layoutCache.xToTime(offset.width, scale: layoutCache.timeScale)
        let newStartTime = CMTimeAdd(clip.timeRangeInTimeline.start, timeOffset)

        // Calculate snap points for current position
        let snapPoints = snappingCoordinator.calculateSnapPoints(
            for: clip,
            in: tracks,
            excluding: [clip.id],
            playheadPosition: playheadPosition
        )

        // Find snap within tolerance
        if let snapResult = snappingCoordinator.findSnapPoint(
            for: newStartTime,
            snapPoints: snapPoints
        ) {
            // Apply snap
            let snappedPosition = snappingCoordinator.applySnapOffset(
                to: newStartTime,
                snapResult: snapResult
            )

            // Update snap indicator
            let snapX = layoutCache.timeToX(snappedPosition, scale: layoutCache.timeScale)
            snapIndicator = SnapIndicator(
                position: snapX,
                snapType: snapResult.snapPoint.type,
                description: snapResult.snapPoint.source ?? ""
            )

            // Update offset to reflect snap
            let snappedOffsetX = layoutCache.timeToX(
                CMTimeSubtract(snappedPosition, clip.timeRangeInTimeline.start),
                scale: layoutCache.timeScale
            )
            dragOffset = CGSize(width: snappedOffsetX, height: offset.height)

        } else {
            // No snap, clear indicator
            snapIndicator = nil
        }

        // Post drag update notification
        NotificationCenter.default.post(
            name: .timelineClipDragging,
            object: self,
            userInfo: ["clipID": clip.id, "position": position, "snapped": snapIndicator != nil]
        )
    }
}
```

- [ ] **Step 4: Implement endDrag method**

```swift
extension TimelineViewModel {
    /// Ends drag operation and commits move
    func endDrag() {
        guard dragState.isDragging, let clip = draggedClip else {
            return
        }

        dragState = .committing

        // Check if position actually changed
        if dragOffset != .zero {
            // Calculate new time range
            let timeOffset = layoutCache.xToTime(dragOffset.width, scale: layoutCache.timeScale)
            let newStartTime = CMTimeAdd(clip.timeRangeInTimeline.start, timeOffset)
            let newTimeRange = CMTimeRange(
                start: newStartTime,
                duration: clip.timeRangeInTimeline.duration
            )

            // Execute move operation through undo/redo coordinator
            let result = undoRedoCoordinator.executeMoveOperation(
                clip: clip,
                newTimeRange: newTimeRange,
                newTrackID: originalTrackID
            )

            switch result {
            case .success:
                // Invalidate layout cache for moved clip
                layoutCache.invalidate(clipID: clip.id)

                // Post success notification
                NotificationCenter.default.post(
                    name: .timelineClipMoved,
                    object: self,
                    userInfo: ["clipID": clip.id]
                )

            case .failure(let error):
                // Revert to original position on error
                layoutCache.invalidate(clipID: clip.id)

                // Post error notification
                NotificationCenter.default.post(
                    name: .timelineDragCancelled,
                    object: self,
                    userInfo: ["clipID": clip.id, "error": error.description]
                )
            }
        }

        // Clear drag state
        draggedClip = nil
        dragOffset = .zero
        snapIndicator = nil
        dragState = .idle

        // Post drag ended notification
        NotificationCenter.default.post(
            name: .timelineDragEnded,
            object: self
        )
    }
}
```

```swift
extension TimelineViewModel {
    /// Cancels the current drag operation
    func cancelDrag() {
        guard dragState.canCancel else {
            return
        }

        // Clear drag state without committing
        if let clip = draggedClip {
            layoutCache.invalidate(clipID: clip.id)
        }

        draggedClip = nil
        dragOffset = .zero
        snapIndicator = nil
        dragState = .cancelled

        // Post cancellation notification
        NotificationCenter.default.post(
            name: .timelineDragCancelled,
            object: self
        )
    }
}
```

```swift
import SwiftUI

struct SnapIndicatorView: View {
    /// Snap indicator data
    let indicator: SnapIndicator?

    /// Height of the timeline (indicator spans full height)
    let timelineHeight: CGFloat

    var body: some View {
        if let indicator = indicator {
            Canvas { context, size in
                // Draw snap line
                context.stroke(
                    Path { path in
                        path.move(to: CGPoint(x: indicator.position, y: 0))
                        path.addLine(to: CGPoint(x: indicator.position, y: timelineHeight))
                    },
                    with: .color(.blue),
                    lineWidth: 2
                )

                // Draw snap indicator circle at top
                let circlePath = Path { path in
                    path.addEllipse(in: CGRect(
                        x: indicator.position - 6,
                        y: 4,
                        width: 12,
                        height: 12
                    ))
                }

                context.fill(circlePath, with: .color(.blue))

                // Draw snap label
                let text = Text(indicator.description)
                    .font(.system(size: 10, weight: .medium))
                    .foregroundColor(.blue)

                context.draw(
                    text,
                    at: CGPoint(x: indicator.position + 8, y: 20)
                )
            }
        }
    }
}
```

```swift
import SwiftUI

struct TimelineView: View {
    /// View model for timeline
    @State private var viewModel: TimelineViewModel

    /// Creates a new timeline view
    /// - Parameter editorState: The editor state
    init(editorState: EditorState) {
        self._viewModel = State(initialValue: TimelineViewModel(editorState: editorState))
    }

    var body: some View {
        VStack(spacing: 0) {
            // Time ruler
            TimeRulerView(
                visibleTimeRange: viewModel.visibleTimeRange,
                playheadPosition: viewModel.playheadPosition,
                timeScale: viewModel.layoutCache.timeScale
            )
            .frame(height: 30)

            // Snap indicator overlay
            snapIndicatorOverlay

            // Tracks
            ScrollView(.horizontal) {
                VStack(spacing: 0) {
                    ForEach(viewModel.tracks) { track in
                        ClipTrackView(
                            track: track,
                            layoutCache: Binding(
                                get: { viewModel.layoutCache },
                                set: { _ in }
                            ),
                            selectedClipIDs: viewModel.selectedClipIDs,
                            onClipSelected: { clipID in
                                viewModel.selectClip(clipID)
                            },
                            onClipDragged: { clipID, offset in
                                handleClipDrag(clipID: clipID, offset: offset)
                            }
                        )
                    }
                }
            }
        }
        .onChange(of: viewModel.snapIndicator) { _, _ in
            // Trigger redraw when snap indicator changes
        }
    }

    private var snapIndicatorOverlay: some View {
        SnapIndicatorView(
            indicator: viewModel.snapIndicator,
            timelineHeight: 200 // TODO: Calculate actual timeline height
        )
        .allowsHitTesting(false)
    }

    private func handleClipDrag(clipID: UUID, offset: CGSize) {
        guard let clip = viewModel.tracks.flatMap({ $0.clips }).first(where: { $0.id == clipID }) else {
            return
        }

        // Find clip position in view
        if let track = viewModel.tracks.first(where: { $0.clips.contains(where: { $0.id == clipID }) }) {
            let frame = viewModel.layoutCache.updateLayout(for: clip, in: track)

            // Convert drag offset to position
            let position = CGPoint(x: frame.x + offset.width, y: frame.y + offset.height)

            switch viewModel.dragState {
            case .idle:
                viewModel.startDrag(clip: clip, at: position)
                viewModel.updateDrag(at: position)

            case .dragStarted, .dragging:
                viewModel.updateDrag(at: position)

            default:
                break
            }
        }
    }
}
```

```swift
extension TimelineViewModel {
    /// Handles keyboard shortcuts
    /// - Parameter keyPress: The key that was pressed
    func handleKeyPress(_ keyPress: KeyPress) {
        switch keyPress {
        case .escape:
            cancelDrag()

        case .cmdZ:
            if NSApp.currentEvent?.modifierFlags.contains(.command) ?? false {
                undo()
            }

        case .cmdShiftZ:
            if NSApp.currentEvent?.modifierFlags.contains([.command, .shift]) ?? false {
                redo()
            }

        case .cmdD:
            deselectAll()

        default:
            break
        }
    }

    /// Performs undo operation
    func undo() {
        let result = undoRedoCoordinator.undo()

        switch result {
        case .success:
            // Invalidate all caches to reflect undo state
            layoutCache.invalidateAll()

        case .failure(let error):
            print("Undo failed: \(error.description)")
        }
    }

    /// Performs redo operation
    func redo() {
        let result = undoRedoCoordinator.redo()

        switch result {
        case .success:
            // Invalidate all caches to reflect redo state
            layoutCache.invalidateAll()

        case .failure(let error):
            print("Redo failed: \(error.description)")
        }
    }
}

/// Supported keyboard shortcuts
enum KeyPress {
    case escape
    case cmdZ
    case cmdShiftZ
    case cmdD
}
```

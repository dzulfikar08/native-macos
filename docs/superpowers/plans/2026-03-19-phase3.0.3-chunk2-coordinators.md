# Phase 3.0.3: Timeline UI - Chunk 2 Coordinator Layer Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build three coordinator components (ClipLayoutCache, SnappingCoordinator, UndoRedoCoordinator) that provide business logic for timeline UI, connecting data models from Chunk 1 to future UI views.

**Architecture:** Coordinator pattern with clear separation of concerns - ClipLayoutCache handles layout calculations and caching, SnappingCoordinator manages snap point calculations, UndoRedoCoordinator bridges UI interactions with Phase 3.0.2's ClipUndoManager. All coordinators are testable plain Swift classes without UI dependencies.

**Tech Stack:** Swift 5.9, @MainActor for thread safety, CoreGraphics for coordinate calculations, CoreMedia for time conversions, Phase 3.0.1 models (VideoClip, ClipTrack), Phase 3.0.2 undo/redo system (ClipUndoManager, MoveClipOperation), Chunk 1 models (SnapPoint, SnapResult, ClipLayout).

```swift
/// Caches clip layout calculations to avoid recalculating clip positions on every render.
/// Tracks dirty state and provides time ↔ position coordinate conversion.
@MainActor
final class ClipLayoutCache {
    /// Cache of clip layout data keyed by clip ID
    private var layoutCache: [UUID: ClipLayout] = [:]

    /// Hash of each track's clips to detect changes
    private var tracksCacheHash: [TrackType: Int] = [:]

    /// Current viewport width in pixels for coordinate conversion
    private(set) var viewportWidth: CGFloat

    /// Time scale (pixels per second) - controls zoom level
    private(set) var timeScale: CGFloat

    /// Creates a new layout cache with specified viewport dimensions
    /// - Parameters:
    ///   - viewportWidth: Width of visible timeline area in pixels
    ///   - timeScale: Pixels per second (default: 20px = 1 second at 600 timescale)
    init(viewportWidth: CGFloat = 1920, timeScale: CGFloat = 20.0) {
        precondition(viewportWidth > 0, "Viewport width must be positive")
        precondition(timeScale > 0, "Time scale must be positive")

        self.viewportWidth = viewportWidth
        self.timeScale = timeScale
    }
}
```

```swift
extension ClipLayoutCache {
    /// Converts timeline time to x-position in pixels
    /// - Parameters:
    ///   - time: Timeline time to convert
    ///   - scale: Time scale in pixels per second (uses instance default if nil)
    /// - Returns: X-position in pixels from timeline start
    func timeToX(_ time: CMTime, scale: CGFloat? = nil) -> CGFloat {
        let effectiveScale = scale ?? timeScale
        let seconds = CMTimeGetSeconds(time)
        precondition(seconds >= 0, "Cannot convert negative time to x-position")
        return CGFloat(seconds) * effectiveScale
    }

    /// Converts x-position to timeline time
    /// - Parameters:
    ///   - x: X-position in pixels
    ///   - scale: Time scale in pixels per second (uses instance default if nil)
    /// - Returns: Timeline time
    func xToTime(_ x: CGFloat, scale: CGFloat? = nil) -> CMTime {
        let effectiveScale = scale ?? timeScale
        precondition(x >= 0, "Cannot convert negative x-position to time")
        let seconds = Double(x) / Double(effectiveScale)
        return CMTime(seconds: seconds, preferredTimescale: 600)
    }

    /// Converts time duration to width in pixels
    /// - Parameters:
    ///   - duration: Time duration to convert
    ///   - scale: Time scale in pixels per second (uses instance default if nil)
    /// - Returns: Width in pixels
    func durationToWidth(_ duration: CMTime, scale: CGFloat? = nil) -> CGFloat {
        let effectiveScale = scale ?? timeScale
        let seconds = CMTimeGetSeconds(duration)
        precondition(seconds >= 0, "Cannot convert negative duration to width")
        return CGFloat(seconds) * effectiveScale
    }
}
```

```swift
extension ClipLayoutCache {
    /// Marks a clip as dirty, requiring layout recalculation
    /// - Parameter clipID: ID of the clip to invalidate
    func invalidate(clipID: UUID) {
        layoutCache[clipID]?.isDirty = true
    }

    /// Marks all clips in a track as dirty
    /// - Parameter track: Track whose clips should be invalidated
    func invalidate(track: ClipTrack) {
        for clip in track.clips {
            invalidate(clipID: clip.id)
        }
    }

    /// Marks all clips as dirty, requiring full recalculation
    func invalidateAll() {
        for clipID in layoutCache.keys {
            layoutCache[clipID]?.isDirty = true
        }
    }

    /// Removes cache entries for deleted clips
    /// - Parameter clipIDs: Set of clip IDs to remove from cache
    func cleanup(_ clipIDs: Set<UUID>) {
        for clipID in clipIDs {
            layoutCache.removeValue(forKey: clipID)
        }
    }
}
```

```swift
extension ClipLayoutCache {
    /// Updates layout for a clip, caching the result
    /// - Parameters:
    ///   - clip: Clip to calculate layout for
    ///   - track: Track containing the clip
    ///   - viewportWidth: Viewport width (uses instance default if nil)
    ///   - timeScale: Time scale (uses instance default if nil)
    /// - Returns: Calculated frame for the clip
    @discardableResult
    func updateLayout(
        for clip: VideoClip,
        in track: ClipTrack,
        viewportWidth: CGFloat? = nil,
        timeScale: CGFloat? = nil
    ) -> CGRect {
        let effectiveViewportWidth = viewportWidth ?? self.viewportWidth
        let effectiveTimeScale = timeScale ?? self.timeScale

        // Calculate x-position from timeline start time
        let x = self.timeToX(clip.timeRangeInTimeline.start, scale: effectiveTimeScale)

        // Calculate width from timeline duration
        let width = self.durationToWidth(clip.timeRangeInTimeline.duration, scale: effectiveTimeScale)

        // Calculate y-position from track position (simplified - will be enhanced in UI chunk)
        let trackIndex = 0 // TODO: Calculate actual track index
        let trackHeight: CGFloat = 60 // Default track height
        let y = CGFloat(trackIndex) * trackHeight

        let frame = CGRect(x: x, y: y, width: width, height: trackHeight)

        // Update cache
        let layout = ClipLayout(clipID: clip.id, frame: frame, timeRange: clip.timeRangeInTimeline, isDirty: false)
        layoutCache[clip.id] = layout

        return frame
    }

    /// Updates the time scale (zoom level)
    /// - Parameter scale: New time scale in pixels per second
    /// - Note: Invalidates all cached layouts when time scale changes
    func setTimeScale(_ scale: CGFloat) {
        precondition(scale > 0, "Time scale must be positive")
        timeScale = scale
        invalidateAll()
    }

    /// Updates the viewport width
    /// - Parameter width: New viewport width in pixels
    func setViewportWidth(_ width: CGFloat) {
        precondition(width > 0, "Viewport width must be positive")
        viewportWidth = width
        // Don't invalidate layouts - they remain valid at same time scale
    }
}
```

```swift
/// Calculates snap points and finds snap targets during drag operations.
/// Supports snapping to clip edges, track boundaries, playhead, time increments, and markers.
@MainActor
final class SnappingCoordinator {
    /// Time distance within which snapping activates (default: 0.5 seconds)
    var snapTolerance: TimeInterval

    /// Whether to snap to clip start/end edges (default: true)
    var snapToClipEdges: Bool

    /// Whether to snap to track boundaries (default: true)
    var snapToTrackBoundaries: Bool

    /// Whether to snap to playhead position (default: true)
    var snapToPlayhead: Bool

    /// Whether to snap to regular time intervals (default: true)
    var snapToTimeIncrements: Bool

    /// Time increment for snapping (default: 1.0 second)
    var timeIncrement: TimeInterval

    /// Whether to snap to chapter markers (default: true)
    var snapToMarkers: Bool

    /// Creates a new snapping coordinator with default settings
    /// - Parameters:
    ///   - snapTolerance: Time tolerance for snapping (default: 0.5s)
    ///   - timeIncrement: Time increment for snapping (default: 1.0s)
    init(snapTolerance: TimeInterval = 0.5, timeIncrement: TimeInterval = 1.0) {
        precondition(snapTolerance > 0, "Snap tolerance must be positive")
        precondition(timeIncrement > 0, "Time increment must be positive")

        self.snapTolerance = snapTolerance
        self.timeIncrement = timeIncrement
        self.snapToClipEdges = true
        self.snapToTrackBoundaries = true
        self.snapToPlayhead = true
        self.snapToTimeIncrements = true
        self.snapToMarkers = true
    }
}
```

```swift
extension SnappingCoordinator {
    /// Calculates all snap points for a clip given current timeline state
    /// - Parameters:
    ///   - clip: Clip being dragged
    ///   - tracks: All tracks in the timeline
    ///   - excluding: Clip IDs to exclude from snap calculation (e.g., the dragged clip itself)
    ///   - playheadPosition: Current playhead position
    ///   - markers: Optional chapter markers
    /// - Returns: Array of snap points
    func calculateSnapPoints(
        for clip: VideoClip,
        in tracks: [ClipTrack],
        excluding clipIDs: Set<UUID> = [],
        playheadPosition: CMTime? = nil,
        markers: [ChapterMarker] = []
    ) -> [SnapPoint] {
        var snapPoints: [SnapPoint] = []

        // Add track boundary snap points (time zero for each track)
        if snapToTrackBoundaries {
            for track in tracks {
                snapPoints.append(SnapPoint(position: .zero, type: .trackBoundary, source: track.name))
            }
        }

        // Add clip edge snap points
        if snapToClipEdges {
            for track in tracks {
                for otherClip in track.clips {
                    // Exclude specified clips (e.g., the clip being dragged)
                    if clipIDs.contains(otherClip.id) {
                        continue
                    }

                    // Add start position
                    snapPoints.append(SnapPoint(
                        position: otherClip.timeRangeInTimeline.start,
                        type: .clipEdge,
                        source: "\(otherClip.name) start"
                    ))

                    // Add end position
                    snapPoints.append(SnapPoint(
                        position: otherClip.timeRangeInTimeline.end,
                        type: .clipEdge,
                        source: "\(otherClip.name) end"
                    ))
                }
            }
        }

        // Add playhead snap point
        if snapToPlayhead, let playhead = playheadPosition {
            snapPoints.append(SnapPoint(position: playhead, type: .playhead, source: "Playhead"))
        }

        // Add time increment snap points
        if snapToTimeIncrements {
            let duration = CMTimeAdd(clip.timeRangeInTimeline.start, clip.timeRangeInTimeline.duration)
            let maxTime = CMTimeGetSeconds(duration) + 10 // Buffer beyond clip
            var currentTime = 0.0
            while currentTime <= maxTime {
                snapPoints.append(SnapPoint(
                    position: CMTime(seconds: currentTime, preferredTimescale: 600),
                    type: .timeIncrement,
                    source: "\(Int(currentTime))s"
                ))
                currentTime += timeIncrement
            }
        }

        // Add marker snap points
        if snapToMarkers {
            for marker in markers {
                snapPoints.append(SnapPoint(
                    position: marker.timestamp,
                    type: .marker,
                    source: marker.title
                ))
            }
        }

        return snapPoints
    }
}
```

```swift
extension SnappingCoordinator {
    /// Finds the closest snap point within tolerance
    /// - Parameters:
    ///   - position: Position to find snap for (in timeline time)
    ///   - snapPoints: Available snap points
    /// - Returns: Snap result if snap found within tolerance, nil otherwise
    func findSnapPoint(for position: CMTime, snapPoints: [SnapPoint]) -> SnapResult? {
        var closestSnap: SnapPoint?
        var closestDistance: TimeInterval = .infinity

        for snapPoint in snapPoints {
            let distance = abs(CMTimeGetSeconds(position) - CMTimeGetSeconds(snapPoint.position))

            if distance < closestDistance {
                closestDistance = distance
                closestSnap = snapPoint
            }
        }

        // Check if closest snap is within tolerance
        if let snap = closestSnap, closestDistance <= snapTolerance {
            return SnapResult(
                snapPoint: snap,
                originalPosition: position,
                snappedPosition: snap.position
            )
        }

        return nil
    }
}
```

```swift
extension UndoRedoCoordinator {
    /// Executes a clip move operation through the undo manager
    /// - Parameters:
    ///   - clip: Clip to move
    ///   - newTimeRange: New time range for the clip
    ///   - newTrackID: Optional new track ID (nil keeps same track)
    /// - Returns: Result indicating success or failure
    func executeMoveOperation(
        clip: VideoClip,
        newTimeRange: CMTimeRange,
        newTrackID: UUID? = nil
    ) -> Result<Void, ClipError> {
        // Prevent recursive operations
        guard !isUndoingOrRedoing else {
            return .failure(.operationFailed("Cannot execute operation during undo/redo"))
        }

        // Create move operation
        let operation = MoveClipOperation(
            clipID: clip.id,
            originalTimeRange: clip.timeRangeInTimeline,
            originalTrackID: clip.trackID,
            newTimeRange: newTimeRange,
            newTrackID: newTrackID ?? clip.trackID,
            clipManager: nil // Will be set by undo manager
        )

        // Execute through undo manager
        do {
            try undoManager.execute(operation)
            return .success(())
        } catch {
            return .failure(error as? ClipError ?? .operationFailed("Unknown error"))
        }
    }
}
```

```swift
extension UndoRedoCoordinator {
    /// Performs undo operation
    /// - Returns: Result indicating success or failure
    func undo() -> Result<Void, ClipError> {
        guard !isUndoingOrRedoing else {
            return .failure(.operationFailed("Already undoing or redoing"))
        }

        isUndoingOrRedoing = true
        defer { isUndoingOrRedoing = false }

        do {
            try undoManager.undo()
            return .success(())
        } catch {
            return .failure(error as? ClipError ?? .operationFailed("Undo failed"))
        }
    }

    /// Performs redo operation
    /// - Returns: Result indicating success or failure
    func redo() -> Result<Void, ClipError> {
        guard !isUndoingOrRedoing else {
            return .failure(.operationFailed("Already undoing or redoing"))
        }

        isUndoingOrRedoing = true
        defer { isUndoingOrRedoing = false }

        do {
            try undoManager.redo()
            return .success(())
        } catch {
            return .failure(error as? ClipError ?? .operationFailed("Redo failed"))
        }
    }

    /// Checks if undo is available
    var canUndo: Bool {
        return undoManager.canUndo
    }

    /// Checks if redo is available
    var canRedo: Bool {
        return undoManager.canRedo
    }
}
```

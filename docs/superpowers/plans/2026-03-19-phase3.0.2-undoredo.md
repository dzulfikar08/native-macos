**Goal:** Implement a comprehensive undo/redo system that wraps all editor operations in reversible command objects, enabling professional-grade editing workflows.

**Architecture:** Command pattern with ClipUndoManager managing operation history. Each editor operation (split, trim, move, delete, duplicate, speed, mode switch, markers, effects) wrapped in dedicated operation class that captures state before modification for undo capability. Smart coalescing merges rapid repeated operations, hybrid history limits prevent unbounded memory growth.

**Tech Stack:** Swift 5.9, @MainActor for thread safety, XCTest, existing OpenScreen patterns (BaseClipOperation, ClipManager, EditorState, notification-based updates)

```swift
import Foundation

/// Manages undo/redo history for editor operations
@MainActor
class ClipUndoManager {
    weak var editorState: EditorState?

    // History management
    private(set) var undoStack: [ClipOperation] = []
    private(set) var redoStack: [ClipOperation] = []

    // Configuration
    var coalescingStrategy: CoalescingStrategy = .smart
    var historyLimit: HistoryLimit = .hybrid(maxOps: 100, timeWindow: 600)

    init(editorState: EditorState) {
        self.editorState = editorState
    }

    // Query state
    var canUndo: Bool { !undoStack.isEmpty }
    var canRedo: Bool { !redoStack.isEmpty }

    // Descriptions for UI
    var undoDescription: String? { undoStack.last?.description }
    var redoDescription: String? { redoStack.last?.description }

    /// Execute an operation and add to undo history
    func executeOperation(_ operation: ClipOperation) throws {
        try operation.execute()
        undoStack.append(operation)
        redoStack.removeAll()
        enforceHistoryLimit()
        postUndoChangeNotification()
    }

    /// Undo the last operation
    func undo() throws {
        guard let operation = undoStack.last else {
            throw UndoError.nothingToUndo
        }

        try operation.undo()
        undoStack.removeLast()
        redoStack.append(operation)
        postUndoChangeNotification()
    }

    /// Redo the last undone operation
    func redo() throws {
        guard let operation = redoStack.last else {
            throw UndoError.nothingToRedo
        }

        try operation.redo()
        redoStack.removeLast()
        undoStack.append(operation)
        postUndoChangeNotification()
    }

    /// Clear all undo/redo history
    func clearHistory() {
        undoStack.removeAll()
        redoStack.removeAll()
        postUndoChangeNotification()
    }

    // MARK: - Private

    private func enforceHistoryLimit() {
        switch historyLimit {
        case .unlimited:
            return
        case .fixedCount(let maxOps):
            while undoStack.count > maxOps {
                undoStack.removeFirst()
            }
        case .timeWindow(let window):
            let cutoff = Date().addingTimeInterval(-window)
            undoStack.removeAll { op in op.timestamp < cutoff }
        case .hybrid(let maxOps, let timeWindow):
            while undoStack.count > maxOps {
                undoStack.removeFirst()
            }
            let cutoff = Date().addingTimeInterval(-timeWindow)
            undoStack.removeAll { op in op.timestamp < cutoff }
        }
    }

    private func postUndoChangeNotification() {
        NotificationCenter.default.post(
            name: .undoStackDidChange,
            object: self,
            userInfo: [
                canUndoKey: canUndo,
                canRedoKey: canRedo
            ]
        )
    }
}

// MARK: - Undo Errors

enum UndoError: LocalizedError {
    case nothingToUndo
    case nothingToRedo

    var errorDescription: String? {
        switch self {
        case .nothingToUndo:
            return "Nothing to undo"
        case .nothingToRedo:
            return "Nothing to redo"
        }
    }
}
```

```swift
import Foundation
import CoreMedia

/// Splits a clip into two clips at a specified time point
class SplitClipOperation: BaseClipOperation {
    private let clipID: UUID
    private let splitTime: CMTime

    // State captured before modification
    private let originalClipState: ClipStateSnapshot

    init(clipID: UUID, splitTime: CMTime, editorState: EditorState, clipManager: ClipManager) {
        self.clipID = clipID
        self.splitTime = splitTime

        // Capture state before modification
        guard let clip = clipManager.findClip(id: clipID),
              let track = clipManager.findTrack(id: clip.trackID) else {
            fatalError("Clip not found during operation initialization")
        }

        self.originalClipState = ClipStateSnapshot(
            clipID: clip.id,
            name: clip.name,
            trackID: clip.trackID,
            timeRangeInSource: clip.timeRangeInSource,
            timeRangeInTimeline: clip.timeRangeInTimeline,
            trackClipCount: track.clips.count
        )

        super.init(
            description: "Split Clip: \(clip.name) at \(CMTimeGetSeconds(splitTime))s",
            editorState: editorState,
            clipManager: clipManager
        )
    }

    func execute() throws {
        try clipManager?.splitClip(clipID: clipID, at: splitTime)
    }

    func undo() throws {
        // Find the two split clips
        guard let clipManager = clipManager,
              let state = editorState else {
            throw UndoError.invalidState
        }

        // Find track
        guard let track = clipManager.findTrack(id: originalClipState.trackID) else {
            throw ClipError.trackNotFound
        }

        // Remove both split clips (they have names with "(L)" and "(R)" suffixes)
        let leftClipName = "\(originalClipState.name) (L)"
        let rightClipName = "\(originalClipState.name) (R)"

        track.clips.removeAll { clip in
            clip.name == leftClipName || clip.name == rightClipName
        }

        // Re-create original clip
        let restoredClip = VideoClip(
            id: originalClipState.clipID,
            name: originalClipState.name,
            asset: originalClipState.asset,
            timeRangeInSource: originalClipState.timeRangeInSource,
            timeRangeInTimeline: originalClipState.timeRangeInTimeline,
            trackID: originalClipState.trackID
        )

        track.addClip(restoredClip)
    }

    func redo() throws {
        try execute()
    }
}

// MARK: - State Snapshot

private struct ClipStateSnapshot {
    let clipID: UUID
    let name: String
    let trackID: UUID
    let timeRangeInSource: CMTimeRange
    let timeRangeInTimeline: CMTimeRange
    let trackClipCount: Int

    var asset: AVAsset {
        // In production, would store URL and recreate
        // For now, we'll find the clip by ID in the track
        fatalError("Asset restoration not implemented")
    }
}
```

```swift
        track.removeClip(id: dupID)
```

```swift
import Foundation
```

```swift
/// Deletes a chapter marker from the timeline
class DeleteMarkerOperation: BaseClipOperation {
    private let marker: ChapterMarker
    private let markerIndex: Int

    init(marker: ChapterMarker, editorState: EditorState) {
        self.marker = marker

        // Capture index for restoration
        guard let index = editorState.chapterMarkers.firstIndex(where: { $0.id == marker.id }) else {
            fatalError("Marker not found during operation initialization")
        }

        self.markerIndex = index

        super.init(
            description: "Delete Marker: \(marker.name)",
            editorState: editorState,
            clipManager: ClipManager(editorState: editorState)
        )
    }

    func execute() throws {
        guard let state = editorState else {
            throw UndoError.invalidState
        }

        state.chapterMarkers.removeAll { $0.id == marker.id }
    }

    func undo() throws {
        guard let state = editorState else {
            throw UndoError.invalidState
        }

        state.chapterMarkers.insert(marker, at: markerIndex)
    }

    func redo() throws {
        try execute()
    }
}
```

```swift
import Foundation
import CoreMedia

/// Applies an effect to a clip
class ApplyEffectOperation: BaseClipOperation {
    private let clipID: UUID
    private let effect: VideoEffect

    // State captured before modification
    private let originalEffectCount: Int

    init(clipID: UUID, effect: VideoEffect, editorState: EditorState) {
        self.clipID = clipID
        self.effect = effect

        // Capture state
        guard let clip = editorState.clipTracks.first?.clips.first(where: { $0.id == clipID }) else {
            fatalError("Clip not found during operation initialization")
        }

        self.originalEffectCount = clip.videoEffects.count

        super.init(
            description: "Apply Effect: \(effect.type) to clip",
            editorState: editorState,
            clipManager: ClipManager(editorState: editorState)
        )
    }

    func execute() throws {
        guard let clip = findClip() else {
            throw ClipError.clipNotFound
        }

        clip.videoEffects.append(effect)
    }

    func undo() throws {
        guard let clip = findClip() else {
            throw ClipError.clipNotFound
        }

        if clip.videoEffects.count > originalEffectCount {
            clip.videoEffects.removeLast()
        }
    }

    func redo() throws {
        try execute()
    }

    private func findClip() -> VideoClip? {
        // Find clip across all tracks
        for track in editorState?.clipTracks ?? [] {
            if let clip = track.clips.first(where: { $0.id == clipID }) {
                return clip
            }
        }
        return nil
    }
}
```

```swift
/// Removes an effect from a clip
class RemoveEffectOperation: BaseClipOperation {
    private let clipID: UUID
    private let effectIndex: Int

    // State captured before modification
    private let removedEffect: VideoEffect

    init(clipID: UUID, effectIndex: Int, editorState: EditorState) {
        self.clipID = clipID
        self.effectIndex = effectIndex

        // Capture state
        guard let clip = editorState.clipTracks.first?.clips.first(where: { $0.id == clipID }),
              effectIndex < clip.videoEffects.count else {
            fatalError("Clip or effect not found during operation initialization")
        }

        self.removedEffect = clip.videoEffects[effectIndex]

        super.init(
            description: "Remove Effect: \(removedEffect.type) from clip",
            editorState: editorState,
            clipManager: ClipManager(editorState: editorState)
        )
    }

    func execute() throws {
        guard let clip = findClip() else {
            throw ClipError.clipNotFound
        }

        guard effectIndex < clip.videoEffects.count else {
            throw UndoError.invalidState
        }

        clip.videoEffects.remove(at: effectIndex)
    }

    func undo() throws {
        guard let clip = findClip() else {
            throw ClipError.clipNotFound
        }

        clip.videoEffects.insert(removedEffect, at: effectIndex)
    }

    func redo() throws {
        try execute()
    }

    private func findClip() -> VideoClip? {
        for track in editorState?.clipTracks ?? [] {
            if let clip = track.clips.first(where: { $0.id == clipID }) {
                return clip
            }
        }
        return nil
    }
}
```

    let name: String

        folder: String = "",

        transitionType: TransitionType,
        parameters: TransitionParameters,

        self.name = name
        self.folder = folder
        self.isFavorite = isFavorite
        self.isBuiltIn = isBuiltIn
        self.transitionType = transitionType

        self.duration = duration
    }

    /// Creates a TransitionClip from this preset
    func makeTransition(
        leadingClipID: UUID,
        trailingClipID: UUID
    ) -> TransitionClip {
        TransitionClip(
            type: transitionType,
            duration: duration,
            leadingClipID: leadingClipID,
            trailingClipID: trailingClipID,
            parameters: parameters,
            isEnabled: true
        )
    }

// Sources/native-macos/Transitions/Presets/TransitionPresetStorage.swift
import Foundation
import CoreImage
import CoreMedia

/// Persists custom presets to disk
struct TransitionPresetStorage {

    private let presetsDirectory: URL

    init(directory: URL? = nil) {
        if let directory = directory {
            presetsDirectory = directory
        } else {
            let supportDir = FileManager.default.urls(for: .applicationSupportDirectory, in: .userDomainMask)[0]
            presetsDirectory = supportDir.appendingPathComponent("OpenScreen/TransitionPresets", isDirectory: true)

            // Create directory if needed
            try? FileManager.default.createDirectory(at: presetsDirectory, withIntermediateDirectories: true, attributes: nil)
        }
    }

    /// Load all custom presets from disk
    func loadCustomPresets() throws -> [TransitionPreset] {
        let files = try FileManager.default.contentsOfDirectory(at: presetsDirectory, includingPropertiesForKeys: nil)

        var presets: [TransitionPreset] = []
        for file in files where file.pathExtension == "json" {
            guard file.lastPathComponent != ".DS_Store" else { continue }
            let data = try Data(contentsOf: file)
            let preset = try JSONDecoder().decode(TransitionPreset.self, from: data)
            presets.append(preset)
        }

        return presets
    }

    /// Save a preset to disk
    func savePreset(_ preset: TransitionPreset) throws {
        guard !preset.isBuiltIn else {
            throw PresetError.cannotModifyBuiltIn
        }

        let encoder = JSONEncoder()
        encoder.outputFormatting = .prettyPrinted
        let data = try encoder.encode(preset)

        // Sanitize filename: percent-encode name, add folder prefix if present
        let folderPrefix = preset.folder.isEmpty ? "" : "\(preset.folder) "
        let sanitizedName = preset.name.addingPercentEncoding(withAllowedCharacters: .alphanumerics) ?? preset.name
        let fileName = "\(folderPrefix)\(sanitizedName).json"
        let fileURL = presetsDirectory.appendingPathComponent(fileName)

        try data.write(to: fileURL)
    }

    /// Delete a preset from disk (including thumbnail cleanup)
    func deletePreset(_ preset: TransitionPreset) throws {
        guard !preset.isBuiltIn else {
            throw PresetError.cannotModifyBuiltIn
        }

        // Delete preset JSON file
        let folderPrefix = preset.folder.isEmpty ? "" : "\(preset.folder) "
        let sanitizedName = preset.name.addingPercentEncoding(withAllowedCharacters: .alphanumerics) ?? preset.name
        let fileName = "\(folderPrefix)\(sanitizedName).json"
        let fileURL = presetsDirectory.appendingPathComponent(fileName)

        try FileManager.default.removeItem(at: fileURL)

        // Delete thumbnail if exists
        try? deleteThumbnail(for: preset)
    }

    /// Save thumbnail for a preset
    func saveThumbnail(_ image: CIImage, for preset: TransitionPreset) throws {
        let thumbnailsDir = presetsDirectory.appendingPathComponent("Thumbnails", isDirectory: true)

        // Create thumbnails directory if needed
        try? FileManager.default.createDirectory(at: thumbnailsDir, withIntermediateDirectories: true, attributes: nil)

        // Generate filename
        let folderPrefix = preset.folder.isEmpty ? "" : "\(preset.folder) "
        let sanitizedName = preset.name.addingPercentEncoding(withAllowedCharacters: .alphanumerics) ?? preset.name
        let fileName = "\(folderPrefix)\(sanitizedName).png"
        let fileURL = thumbnailsDir.appendingPathComponent(fileName)

        // Convert CIImage to CGImage then to PNG
        let rep = NSCIImageRep(ciImage: image)
        let nsImage = NSImage(size: rep.size)
        nsImage.addRepresentation(rep)

        guard let tiffData = nsImage.tiffRepresentation,
              let bitmap = NSBitmapImageRep(data: tiffData),
              let pngData = bitmap.representation(using: .png, properties: [:]) else {
            throw PresetError.thumbnailGenerationFailed
        }

        try pngData.write(to: fileURL)
    }

    /// Load thumbnail for a preset
    func loadThumbnail(for preset: TransitionPreset) throws -> CIImage? {
        let thumbnailsDir = presetsDirectory.appendingPathComponent("Thumbnails", isDirectory: false)

        let folderPrefix = preset.folder.isEmpty ? "" : "\(preset.folder) "
        let sanitizedName = preset.name.addingPercentEncoding(withAllowedCharacters: .alphanumerics) ?? preset.name
        let fileName = "\(folderPrefix)\(sanitizedName).png"
        let fileURL = thumbnailsDir.appendingPathComponent(fileName)

        guard FileManager.default.fileExists(atPath: fileURL.path) else {
            return nil
        }

        guard let image = NSImage(contentsOf: fileURL),
              let cgImage = image.cgImage(forProposedRect: nil, hints: nil) else {
            return nil
        }

        return CIImage(cgImage: cgImage)
    }

    /// Delete thumbnail for a preset
    private func deleteThumbnail(for preset: TransitionPreset) throws {
        let thumbnailsDir = presetsDirectory.appendingPathComponent("Thumbnails", isDirectory: false)

        let folderPrefix = preset.folder.isEmpty ? "" : "\(preset.folder) "
        let sanitizedName = preset.name.addingPercentEncoding(withAllowedCharacters: .alphanumerics) ?? preset.name
        let fileName = "\(folderPrefix)\(sanitizedName).png"
        let fileURL = thumbnailsDir.appendingPathComponent(fileName)

        try? FileManager.default.removeItem(at: fileURL)
    }
}

// Sources/native-macos/Transitions/Presets/PresetError.swift
import Foundation

enum PresetError: Error, LocalizedError {
    case cannotModifyBuiltIn
    case emptyName
    case reservedName
    case duplicateName(String)
    case saveFailed(reason: String)
    case diskFull
    case invalidFile
    case incompatibleVersion
    case thumbnailGenerationFailed

    var errorDescription: String? {
        switch self {
        case .cannotModifyBuiltIn:
            return "Cannot modify built-in presets"
        case .emptyName:
            return "Preset name cannot be empty"
        case .reservedName:
            return "This name is reserved for built-in presets"
        case .duplicateName(let name):
            return "A preset named '\(name)' already exists"
        case .saveFailed(let reason):
            return "Failed to save preset: \(reason)"
        case .diskFull:
            return "Not enough disk space to save preset"
        case .invalidFile:
            return "The selected file is not a valid preset"
        case .incompatibleVersion:
            return "This preset is from an incompatible version"
        case .thumbnailGenerationFailed:
            return "Failed to generate preset thumbnail"
        }
    }
}

// Sources/native-macos/Transitions/Presets/PresetLibrary.swift
import Foundation
import CoreMedia
import Combine

/// Manages in-memory collection of all presets (built-in + custom)
@MainActor
final class PresetLibrary: ObservableObject {

    /// All presets (built-in + custom)
    private(set) var allPresets: [TransitionPreset] = []

    /// Folder organization
    private(set) var folders: Set<String> = ["My Transitions"]

    private let storage: TransitionPresetStorage

    /// Initialize library with built-in presets
    init(storage: TransitionPresetStorage = TransitionPresetStorage()) {
        self.storage = storage
        loadBuiltInPresets()
    }

    /// Initialize with custom storage (for testing)
    init(storage: TransitionPresetStorage) {
        self.storage = storage
        loadBuiltInPresets()
    }

    // MARK: - Loading

    /// Load built-in presets
    private func loadBuiltInPresets() {
        allPresets = BuiltInPresets.presets
    }

    /// Load custom presets from storage
    func loadCustomPresets() throws {
        let customPresets = try storage.loadCustomPresets()
        allPresets.append(contentsOf: customPresets)

        // Extract folders from custom presets
        let folderNames = Set(customPresets.map { $0.folder })
        folders = folders.union(folderNames).filter { !$0.isEmpty }
    }

    // MARK: - Preset Management

    /// Validate preset name
    func validatePresetName(_ name: String) throws {
        guard !name.isEmpty else {
            throw PresetError.emptyName
        }

        guard !BuiltInPresets.presets.contains(where: { $0.name == name }) else {
            throw PresetError.reservedName
        }

        guard !allPresets.contains(where: { $0.name == name }) else {
            throw PresetError.duplicateName(name)
        }
    }

    /// Save a new custom preset
    func savePreset(
        name: String,
        folder: String,
        transition: TransitionClip,
        isFavorite: Bool
    ) throws {
        try validatePresetName(name)

        let preset = TransitionPreset(
            name: name,
            folder: folder,
            isFavorite: isFavorite,
            isBuiltIn: false,
            transitionType: transition.type,
            parameters: transition.parameters,
            duration: transition.duration
        )

        try storage.savePreset(preset)
        allPresets.append(preset)

        if !folder.isEmpty {
            folders.insert(folder)
        }

        objectWillChange.send()
    }

    /// Delete a custom preset
    func deletePreset(_ preset: TransitionPreset) throws {
        guard !preset.isBuiltIn else {
            throw PresetError.cannotModifyBuiltIn
        }

        try storage.deletePreset(preset)
        allPresets.removeAll { $0.id == preset.id }

        objectWillChange.send()
    }

    /// Update preset metadata
    func updatePreset(_ preset: TransitionPreset, name: String? = nil, folder: String? = nil) throws {
        guard !preset.isBuiltIn else {
            throw Preset.cannotModifyBuiltIn
        }

        guard let index = allPresets.firstIndex(where: { $0.id == preset.id }) else {
            return
        }

        var updated = preset

        if let name = name {
            // Temporarily remove to validate
            allPresets.remove(at: index)
            try validatePresetName(name)
            allPresets.insert(updated, at: index)

            updated = TransitionPreset(
                id: updated.id,
                name: name,
                folder: folder ?? updated.folder,
                isFavorite: updated.isFavorite,
                isBuiltIn: updated.isBuiltIn,
                transitionType: updated.transitionType,
                parameters: updated.parameters,
                duration: updated.duration
            )
        } else if let folder = folder {
            allPresets.remove(at: index)
            updated = TransitionPreset(
                id: updated.id,
                name: updated.name,
                folder: folder,
                isFavorite: updated.isFavorite,
                isBuiltIn: updated.isBuiltIn,
                transitionType: updated.transitionType,
                parameters: updated.parameters,
                duration: updated.duration
            )
            allPresets.insert(updated, at: index)
        }

        // Save updated preset
        try storage.savePreset(updated)

        // Update in-memory copy
        allPresets[index] = updated

        // Update folders if needed
        if let folder = folder, !folder.isEmpty {
            folders.insert(folder)
        }

        objectWillChange.send()
    }

    /// Toggle favorite status
    func toggleFavorite(_ preset: TransitionPreset) throws {
        guard !preset.isBuiltIn else {
            throw PresetError.cannotModifyBuiltIn
        }

        guard let index = allPresets.firstIndex(where: { $0.id == preset.id }) else {
            return
        }

        var updated = preset
        updated = TransitionPreset(
            id: updated.id,
            name: updated.name,
            folder: updated.folder,
            isFavorite: !updated.isFavorite,
            isBuiltIn: updated.isBuiltIn,
            transitionType: updated.transitionType,
            parameters: updated.parameters,
            duration: updated.duration
        )

        allPresets[index] = updated

        try storage.savePreset(updated)
        objectWillChange.send()
    }

    // MARK: - Queries

    /// Get presets in a specific folder
    func presetsInFolder(_ folder: String) -> [TransitionPreset] {
        switch folder {
        case "All":
            return allPresets
        case "Favorites":
            return allPresets.filter { $0.isFavorite }
        default:
            return allPresets.filter { $0.folder == folder }
        }
    }

    /// Get favorite presets
    func favoritePresets() -> [TransitionPreset] {
        return allPresets.filter { $0.isFavorite }
    }

    // MARK: - Import/Export

    /// Import preset from file
    func importPreset(from url: URL) throws {
        let data = try Data(contentsOf: url)
        var preset = try JSONDecoder().decode(TransitionPreset.self, from: data)

        // Mark as not built-in
        preset = TransitionPreset(
            id: preset.id,
            name: preset.name,
            folder: preset.folder.isEmpty ? "Imported" : preset.folder,
            isFavorite: preset.isFavorite,
            isBuiltIn: false,
            transitionType: preset.transitionType,
            parameters: preset.parameters,
            duration: preset.duration
        )

        // Handle name collision
        if allPresets.contains(where: { $0.name == preset.name }) {
            // Auto-rename by appending " (Imported)"
            preset = TransitionPreset(
                id: UUID(),
                name: "\(preset.name) (Imported)",
                folder: preset.folder,
                isFavorite: false,
                isBuiltIn: false,
                transitionType: preset.transitionType,
                parameters: preset.parameters,
                duration: preset.duration
            )
        }

        try storage.savePreset(preset)
        allPresets.append(preset)

        if !preset.folder.isEmpty {
            folders.insert(preset.folder)
        }

        objectWillChange.send()
    }

    /// Export preset to file
    func exportPreset(_ preset: TransitionPreset, to url: URL) throws {
        let encoder = JSONEncoder()
        encoder.outputFormatting = .prettyPrinted
        let data = try encoder.encode(preset)
        try data.write(to: url)
    }

    /// Create new folder
    func createFolder(name: String) {
        folders.insert(name)
        objectWillChange.send()
    }
}

// Sources/native-macos/Transitions/Presets/PresetPreviewRenderer.swift
import Foundation
import CoreImage
import CoreMedia
import AppKit

/// Generates preset preview thumbnails
struct PresetPreviewRenderer {

    static let thumbnailSize = CGSize(width: 128, height: 72)

    private let context: TransitionRenderContext
    private var thumbnailCache: [UUID: CIImage] = [:]

    init(context: TransitionRenderContext = TransitionRenderContext()) {
        self.context = context
    }

    /// Generate thumbnail for a preset
    func generateThumbnail(for preset: TransitionPreset) throws -> CIImage {
        // Check cache first
        if let cached = thumbnailCache[preset.id] {
            return cached
        }

        // Create test frames
        let leadingFrame = createTestFrame(color: .red)
        let trailingFrame = createTestFrame(color: .blue)

        // Get renderer
        guard let renderer = context.renderer(for: preset) else {
            throw PresetError.thumbnailGenerationFailed
        }

        // Render at midpoint
        let progress = 0.5
        let rendered = try? renderer.render(
            sourceBuffer: createPixelBuffer(color: .red),
            targetBuffer: createPixelBuffer(color: .blue),
            progress: progress,
            transition: preset.makeTransition(leadingClipID: UUID(), trailingClipID: UUID())
        )

        guard let outputBuffer = rendered else {
            throw PresetError.thumbnailGenerationFailed
        }

        // Convert to CIImage
        let ciImage = CIImage(cvPixelBuffer: outputBuffer)

        // Scale to thumbnail size
        let scaled = ciImage.transformed(by: CGAffineTransform(scaleX: CGFloat(128) / CGFloat(ciImage.extent.width),
                                                   y: CGFloat(72) / CGFloat(ciImage.extent.height))

        // Crop to thumbnail size
        let cropped = scaled.cropped(to: CGRect(x: 0, y: 0, width: 128, height: 72))

        // Cache
        thumbnailCache[preset.id] = cropped

        return cropped
    }

    /// Load or generate thumbnail (uses cache)
    func thumbnail(for preset: TransitionPreset, storage: TransitionPresetStorage) -> CIImage? {
        // Check disk cache
        if let cached = try? storage.loadThumbnail(for: preset) {
            thumbnailCache[preset.id] = cached
            return cached
        }

        // Generate and save
        guard let generated = try? generateThumbnail(for: preset) else {
            return nil
        }

        try? storage.saveThumbnail(generated, for: preset)

        return generated
    }

    // MARK: - Helper Methods

    private func createTestFrame(color: NSColor) -> CIImage {
        let size = CGSize(width: 1920, height: 1080)
        let rect = CGRect(origin: .zero, size: size)

        let image = CIImage(color: .constant(color.cgColor!.cgColor))
        return image.cropped(to: rect)
    }

    private func createPixelBuffer(color: NSColor) -> CVPixelBuffer {
        let width = 1920
        let height = 1080

        var pixelBuffer: CVPixelBuffer?
        let status = CVPixelBufferCreate(
            kCFAllocatorDefault,
            width,
            height,
            kCVPixelFormatType_32ARGB,
            nil,
            &pixelBuffer
        )

        guard status == kCVReturnSuccess, let buffer = pixelBuffer else {
            fatalError("Failed to create pixel buffer")
        }

        CVPixelBufferLockBaseAddress(buffer, .mutable)
        defer { CVPixelBufferUnlockBaseAddress(buffer, []) }

        let data = CVPixelBufferGetBaseAddress(buffer)
        let rgbColor = color.cgColor?.components ?? [0, 0, 0, 1]

        for y in 0..<height {
            for x in 0..<width {
                let offset = (y * width + x) * 4
                data?[offset] = UInt8(rgbColor[0] * 255)
                data?[offset + 1] = UInt8(rgbColor[1] * 255)
                data?[offset + 2] = UInt8(rgbColor[2] * 255)
                data?[offset + 3] = UInt8(rgbColor[3] * 255)
            }
        }

        return buffer
    }
}

        scrollView.drawsBackground = false
    }

    required init?(coder: NSCoder) {

    private func reloadGrid() {
        // Clear existing views
        presetViews.forEach { $0.removeFromSuperview() }
        presetViews.removeAll()

        // Create card for each preset
        for preset in presets {
            let cardView = PresetCardView(preset: preset)
            cardView.onPresetSelected = onPresetSelected
            cardView.onFavoriteToggled = onFavoriteToggled
            documentView.addSubview(cardView)
            presetViews.append(cardView)
        }

        // Layout cards in grid
        layoutCards()
    }

    private func layoutCards() {
        // Simple grid layout - 3 columns
        let cardSize = NSSize(width: 200, height: 150)
        let spacing: CGFloat = 12
        let columns = 3

        for (index, card) in presetViews.enumerated() {
            let row = index / columns
            let col = index % columns

            let x = CGFloat(col) * (cardSize.width + spacing)
            let y = CGFloat(row) * (cardSize.height + spacing)

            card.frame = NSRect(origin: NSPoint(x: x, y: y), size: cardSize)
        }

        documentView.frame = NSRect(
            origin: .zero,
            size: NSSize(
                width: cardSize.width * CGFloat(columns) + spacing * CGFloat(columns - 1),
                height: ceil(CGFloat(presetViews.count) / CGFloat(columns)) * (cardSize.height + spacing)
            )
        )
    }

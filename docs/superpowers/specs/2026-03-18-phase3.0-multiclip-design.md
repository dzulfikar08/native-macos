# Phase 3.0: Multi-Clip Editing Architecture - Design Specification

**Status:** Approved
**Date:** 2026-03-18
**Authors:** User + Claude Sonnet 4.6
**Related:** Phase 3.1 Video Transitions (depends on this phase)

## Overview

Phase 3.0 establishes the foundational architecture for professional multi-clip editing in OpenScreen. This phase transforms the current single-asset editing model into a dual-mode system that supports both simple single-recording workflows and complex multi-clip compositions while maintaining full backward compatibility.

**Key Goals:**
- Enable professional video editing workflows with multiple clips per timeline
- Preserve backward compatibility with existing single-asset projects
- Support all professional NLE features: multicam, sync bins, compound clips
- Maintain 60fps performance with multiple simultaneous clips
- Enable seamless transitions between single-asset and multi-clip modes

**Timeline:** 8-13 weeks (5 sub-phases)
**Dependencies:** None (foundation for Phase 3.1 Transitions and beyond)

## Section 1: Data Models

### 1.1 VideoClip Model

**Purpose:** Represents a single piece of video content on the timeline, wrapping an AVAsset with timeline positioning.

**Location:** `Sources/native-macos/Timeline/VideoClip.swift`

```swift
import AVFoundation
import Foundation

/// A video clip on the timeline, wrapping an AVAsset with positioning information
/// Note: Not Sendable due to AVAsset. All access must remain on @MainActor.
@MainActor
class VideoClip: Identifiable, Observable {
    let id: UUID
    var name: String

    /// The underlying AVAsset (URL or composition)
    /// Note: AVAsset is not Sendable, so VideoClip is @MainActor-bound
    let asset: AVAsset

    /// Portion of the source asset to use (e.g., trim start/end)
    var timeRangeInSource: CMTimeRange

    /// Where the clip appears on the timeline
    var timeRangeInTimeline: CMTimeRange

    /// Which track this clip belongs to
    var trackID: UUID

    /// Clip state
    var isEnabled: Bool = true
    var opacity: Float = 1.0
    var speed: Float = 1.0

    /// Audio properties
    var volume: Float = 1.0
    var isAudioMuted: Bool = false

    /// Effects applied to this clip
    var videoEffects: [VideoEffect] = []
    var audioEffects: [AudioEffect] = []

    /// Clip metadata
    var createdAt: Date
    var modifiedAt: Date

    init(
        id: UUID = UUID(),
        name: String,
        asset: AVAsset,
        timeRangeInSource: CMTimeRange,
        timeRangeInTimeline: CMTimeRange,
        trackID: UUID,
        opacity: Float = 1.0,
        speed: Float = 1.0,
        volume: Float = 1.0
    ) {
        self.id = id
        self.name = name
        self.asset = asset
        self.timeRangeInSource = timeRangeInSource
        self.timeRangeInTimeline = timeRangeInTimeline
        self.trackID = trackID
        self.opacity = opacity
        self.speed = speed
        self.volume = volume
        self.createdAt = Date()
        self.modifiedAt = Date()
    }

    /// Duration of this clip on timeline (accounting for speed)
    var timelineDuration: CMTime {
        let sourceDuration = timeRangeInSource.duration
        return CMTimeMultiplyByFloat64(sourceDuration, multiplier: Float64(1.0 / speed))
    }

    /// Get frame from this clip at timeline time
    func frame(at timelineTime: CMTime) -> CIImage? {
        let clipTime = timelineTimeInSource(timelineTime)
        return extractFrame(from: asset, at: clipTime)
    }

    /// Convert timeline time to source asset time
    func timelineTimeInSource(_ timelineTime: CMTime) -> CMTime {
        let offset = timelineTime - timeRangeInTimeline.start
        let scaledOffset = CMTimeMultiplyByFloat64(offset, multiplier: Float64(speed))
        return timeRangeInSource.start + scaledOffset
    }
}
```

**Validation:**
- `timeRangeInSource` must be within asset duration
- `timeRangeInTimeline.start` must be >= 0
- `speed` must be in range [0.1, 16.0]
- `opacity` must be in range [0.0, 1.0]
- `volume` must be in range [0.0, 2.0]

### 1.2 ClipTrack Model

**Purpose:** Container for ordered clips that share compositing and rendering properties.

**Location:** `Sources/native-macos/Timeline/ClipTrack.swift`

```swift
import Foundation
import AppKit

enum TrackType: String, Codable, Sendable {
    case video
    case audio
    case title
    case effects
}

/// Note: Not Sendable due to @Published clips array. All access must remain on @MainActor.
@MainActor
class ClipTrack: Identifiable, Observable {
    let id: UUID
    var name: String
    var type: TrackType

    /// Ordered clips on this track (sorted by timeRangeInTimeline.start)
    @Published var clips: [VideoClip] = []

    /// Track state
    var isEnabled: Bool = true
    var isLocked: Bool = false
    var isVisible: Bool = true

    /// Compositing properties (for video tracks)
    var opacity: Float = 1.0
    var compositingMode: CICompositingMode = .normal

    /// Z-index for rendering order (lower = renders first/behind)
    var zIndex: Int = 0

    /// Track height in timeline UI
    var height: Double = 80.0

    init(
        id: UUID = UUID(),
        name: String,
        type: TrackType,
        zIndex: Int = 0
    ) {
        self.id = id
        self.name = name
        self.type = type
        self.zIndex = zIndex
    }

    /// Add clip to track (maintains sorted order)
    func addClip(_ clip: VideoClip) {
        clips.append(clip)
        sortClips()
    }

    /// Remove clip from track
    func removeClip(id: UUID) {
        clips.removeAll { $0.id == id }
    }

    /// Find clip at timeline time
    func clip(at timelineTime: CMTime) -> VideoClip? {
        clips.first { clip in
            timelineTime >= clip.timeRangeInTimeline.start &&
            timelineTime < clip.timeRangeInTimeline.end
        }
    }

    /// Maintain clips sorted by start time
    private func sortClips() {
        clips.sort { $0.timeRangeInTimeline.start < $1.timeRangeInTimeline.start }
    }
}
```

### 2.2 Mode Switching

**Location:** `Sources/native-macos/Shared/Models/EditorState.swift`

```swift
extension EditorState {
    /// Timeline edit mode for this project
    @Published var timelineEditMode: TimelineEditMode = .singleAsset

    /// Multi-clip tracks (only used in .multiClip mode)
    /// Note: ClipTrack is used in multi-clip mode, TimelineTrack is used in single-asset mode
    @Published var clipTracks: [ClipTrack] = []

    /// Clip operation history for undo/redo
    var clipOperations: [ClipOperation] = []
    var redoStack: [ClipOperation] = []

    /// Switch from single-asset to multi-clip mode
    /// - Parameter splitAtMarkers: If true and chapter markers exist, split at marker boundaries. Otherwise creates single clip.
    func switchToMultiClipMode(splitAtMarkers: Bool = true) throws {
        guard timelineEditMode == .singleAsset else {
            throw ClipError.alreadyInMultiClipMode
        }

        guard let currentAsset = asset else {
            throw ClipError.clipNotFound  // Reuse error: no asset loaded
        }

        // Create default video track
        let videoTrack = ClipTrack(
            id: UUID(),
            name: "Video 1",
            type: .video,
            zIndex: 0
        )

        // Convert asset to clips
        if splitAtMarkers && !chapterMarkers.isEmpty {
            // Split at each marker boundary
            let sortedMarkers = chapterMarkers.sorted { $0.time < $1.time }
            var lastTime = CMTime.zero

            for marker in sortedMarkers {
                let clipRange = CMTimeRange(start: lastTime, duration: marker.time - lastTime)

                // Validate range is valid
                guard clipRange.duration > .zero else { continue }

                let clip = VideoClip(
                    name: "Clip \(videoTrack.clips.count + 1)",
                    asset: currentAsset,
                    timeRangeInSource: clipRange,
                    timeRangeInTimeline: clipRange,
                    trackID: videoTrack.id
                )
                videoTrack.addClip(clip)
                lastTime = marker.time
            }

            // Add final clip from last marker to end
            if lastTime < currentAsset.duration {
                let finalRange = CMTimeRange(start: lastTime, end: currentAsset.duration)
                let finalClip = VideoClip(
                    name: "Clip \(videoTrack.clips.count + 1)",
                    asset: currentAsset,
                    timeRangeInSource: finalRange,
                    timeRangeInTimeline: finalRange,
                    trackID: videoTrack.id
                )
                videoTrack.addClip(finalClip)
            }
        } else {
            // Create single clip from entire asset
            let fullRange = CMTimeRange(start: .zero, end: currentAsset.duration)
            let clip = VideoClip(
                name: currentAsset.url?.lastPathComponent ?? "Recording",
                asset: currentAsset,
                timeRangeInSource: fullRange,
                timeRangeInTimeline: fullRange,
                trackID: videoTrack.id
            )
            videoTrack.addClip(clip)
        }

        // Add default audio track if asset has audio
        if currentAsset.tracks(withMediaType: .audio).isEmpty == false {
            let audioTrack = ClipTrack(
                id: UUID(),
                name: "Audio 1",
                type: .audio,
                zIndex: 1
            )
            clipTracks = [videoTrack, audioTrack]
        } else {
            clipTracks = [videoTrack]
        }

        timelineEditMode = .multiClip

        NotificationCenter.default.post(
            name: .timelineEditModeDidChange,
            object: self
        )
    }

    /// Switch from multi-clip back to single-asset mode (destructive warning)
    func switchToSingleAssetMode() throws {
        guard timelineEditMode == .multiClip else {
            throw ClipError.alreadyInSingleAssetMode
        }

        // Warning: This is a destructive operation
        // All clip edits will be lost. User should export composition first.

        // For now, preserve clipTracks data in case user switches back
        // In future, could prompt user to export or save composition

        timelineEditMode = .singleAsset

        NotificationCenter.default.post(
            name: .timelineEditModeDidChange,
            object: self
        )
    }
}
```

**Implementation Notes:**
- `editorState.asset` will be made public (currently private) for dual-mode access
- Chapter markers are preserved when converting to clips (markers → clip boundaries)
- Default track structure: Video track (zIndex 0) + Audio track (zIndex 1) if audio present
- Switching back to single-asset mode is destructive - clips are not converted back to single asset
- Users should export their multi-clip composition before switching back to single-asset mode

```swift
/// Protocol for undoable clip operations
protocol ClipOperation {
    /// Human-readable description of this operation
    var description: String { get }

    /// Execute the operation
    func execute() throws

    /// Undo the operation
    func undo() throws

    /// Redo the operation (default uses execute())
    func redo() throws
}

/// Base class for clip operations with undo/redo support
class BaseClipOperation: ClipOperation {
    let description: String
    weak var editorState: EditorState?
    weak var clipManager: ClipManager?

    init(description: String, editorState: EditorState, clipManager: ClipManager) {
        self.description = description
        self.editorState = editorState
        self.clipManager = clipManager
    }

    func execute() throws {
        fatalError("Subclasses must implement execute()")
    }

    func undo() throws {
        fatalError("Subclasses must implement undo()")
    }

    func redo() throws {
        try execute()
    }
}

/// Example: Split clip operation
class SplitClipOperation: BaseClipOperation {
    let clipID: UUID
    let splitTime: CMTime
    var resultingClipIDs: [UUID] = []

    override func execute() throws {
        try clipManager?.splitClip(clipID: clipID, at: splitTime)
        // Capture resulting clip IDs from notification or return value
    }

    override func undo() throws {
        // Delete the split clips
        for clipID in resultingClipIDs {
            try clipManager?.deleteClip(clipID: clipID, ripple: false)
        }
        // Restore original clip (from saved state)
    }
}
```

**Undo/Redo Stack Management:**

**Location:** `Sources/native-macos/Timeline/UndoManager.swift`

```swift
@MainActor
class ClipUndoManager: ObservableObject {
    weak var editorState: EditorState?

    private var maxStackSize: Int = 100
    private var maxMemoryMB: Int = 50

    func executeOperation(_ operation: ClipOperation) throws {
        // Execute the operation
        try operation.execute()

        // Add to undo stack
        editorState?.clipOperations.append(operation)

        // Clear redo stack (new operation invalidates redo history)
        editorState?.redoStack.removeAll()

        // Enforce stack limits
        cleanupStacks()
    }

    func undo() throws {
        guard let operation = editorState?.clipOperations.last else {
            throw ClipError.clipNotFound  // No operation to undo
        }

        try operation.undo()
        editorState?.clipOperations.removeLast()
        editorState?.redoStack.append(operation)
    }

    func redo() throws {
        guard let operation = editorState?.redoStack.last else {
            throw ClipError.clipNotFound  // No operation to redo
        }

        try operation.redo()
        editorState?.redoStack.removeLast()
        editorState?.clipOperations.append(operation)
    }

    private func cleanupStacks() {
        // Remove old operations if stack exceeds max size
        while let count = editorState?.clipOperations.count, count > maxStackSize {
            editorState?.clipOperations.removeFirst()
        }

        // TODO: Add memory-based cleanup (estimate operation memory usage)
    }

    func clear() {
        editorState?.clipOperations.removeAll()
        editorState?.redoStack.removeAll()
    }
}
```

## Section 4: Rendering Pipeline

### 4.1 Dual-Mode VideoProcessor

**Location:** `Sources/native-macos/Editing/VideoProcessor.swift`

```swift
extension VideoProcessor {
    func renderFrame(at time: CMTime) -> CIImage? {
        switch editorState.timelineEditMode {
        case .singleAsset:
            return renderSingleAssetFrame(at: time)
        case .multiClip:
            return renderMultiClipFrame(at: time)
        }
    }

    private func renderMultiClipFrame(at time: CMTime) -> CIImage? {
        var compositedImage: CIImage?
        let sortedTracks = editorState.clipTracks.sorted { $0.zIndex < $1.zIndex }

        for track in sortedTracks {
            guard track.isEnabled && track.isVisible else { continue }

            let activeClips = track.clips.filter { clip in
                clip.isEnabled &&
                time >= clip.timeRangeInTimeline.start &&
                time < clip.timeRangeInTimeline.end
            }

            for clip in activeClips {
                guard let clipFrame = renderClipFrame(clip, at: time) else { continue }
                let processedFrame = applyClipEffects(to: clipFrame, clip: clip)

                if let existing = compositedImage {
                    compositedImage = composite(processedFrame, over: existing)
                } else {
                    compositedImage = processedFrame
                }
            }
        }

        return compositedImage
    }
}
```

### 4.2 ClipFrameCache Implementation

**Purpose:** LRU cache for clip frames to maintain 60fps performance during multi-clip playback.

**Location:** `Sources/native-macos/Editing/ClipFrameCache.swift`

```swift
import Foundation
import CoreImage

/// LRU cache for clip frames
actor ClipFrameCache {
    private struct CacheKey: Hashable {
        let clipID: UUID
        let timelineTime: CMTime
    }

    private var cache: [CacheKey: CIImage] = [:]
    private var accessOrder: [CacheKey] = []
    private let maxCapacity: Int = 100

    // Statistics for performance monitoring
    private var hitCount: Int = 0
    private var missCount: Int = 0

    /// Get cached frame for clip at timeline time
    func get(clipID: UUID, timelineTime: CMTime) -> CIImage? {
        let key = CacheKey(clipID: clipID, timelineTime: timelineTime)

        if let image = cache[key] {
            // Cache hit - move to end (most recently used)
            hitCount += 1
            accessOrder.removeAll { $0 == key }
            accessOrder.append(key)
            return image
        }

        // Cache miss
        missCount += 1
        return nil
    }

    /// Store frame in cache
    func set(_ image: CIImage, for clipID: UUID, timelineTime: CMTime) {
        let key = CacheKey(clipID: clipID, timelineTime: timelineTime)

        // Evict LRU entry if at capacity
        if cache.count >= maxCapacity {
            evictLRU()
        }

        cache[key] = image
        accessOrder.append(key)
    }

    /// Invalidate all cached frames for a specific clip
    func invalidate(clipID: UUID) {
        let keysToRemove = cache.keys.filter { $0.clipID == clipID }
        for key in keysToRemove {
            cache.removeValue(forKey: key)
            accessOrder.removeAll { $0 == key }
        }
    }

    /// Clear entire cache
    func clear() {
        cache.removeAll()
        accessOrder.removeAll()
        hitCount = 0
        missCount = 0
    }

    /// Evict least recently used entry
    private func evictLRU() {
        guard let lruKey = accessOrder.first else { return }
        cache.removeValue(forKey: lruKey)
        accessOrder.removeFirst()
    }

    /// Cache hit rate (0.0 to 1.0)
    var hitRate: Double {
        let total = hitCount + missCount
        return total > 0 ? Double(hitCount) / Double(total) : 0.0
    }

    /// Estimated memory usage (assuming 5MB per 1080p frame)
    var estimatedMemoryMB: Int {
        return cache.count * 5
    }
}
```

**Integration with VideoProcessor:**

```swift
extension VideoProcessor {
    private let clipFrameCache = ClipFrameCache()

    private func renderClipFrame(_ clip: VideoClip, at time: CMTime) -> CIImage? {
        // Check cache first
        if let cached = await clipFrameCache.get(clipID: clip.id, timelineTime: time) {
            return cached
        }

        // Cache miss - render frame
        guard let frame = renderClipFrameUncached(clip, at: time) else {
            return nil
        }

        // Store in cache
        await clipFrameCache.set(frame, for: clip.id, timelineTime: time)

        return frame
    }

    private func renderClipFrameUncached(_ clip: VideoClip, at time: CMTime) -> CIImage? {
        let clipTime = clip.timelineTimeInSource(time)
        return extractFrame(from: clip.asset, at: clipTime)
    }
}
```

**Cache Invalidation Triggers:**
- Clip is modified (trim, speed change, effect added): `clipFrameCache.invalidate(clipID: clip.id)`
- Clip is deleted: `clipFrameCache.invalidate(clipID: clip.id)`
- Cache exceeds memory limit: Automatic LRU eviction
- User clears cache: `clipFrameCache.clear()`

**Performance Targets:**
- Cache hit rate: >80% during normal playback
- Memory usage: <500MB at 1080p (100 frames × 5MB)
- Cache lookup: <1ms
- Frame render with cache: <2ms total

**Layout Algorithm:**

```swift
struct MultiClipTimelineView: View {
    @ObservedObject var editorState: EditorState

    // Timeline scale: pixels per second (e.g., 50px = 1 second)
    @State private var pixelsPerSecond: Double = 50.0
    @State private var contentOffset: CGPoint = .zero

    var body: some View {
        GeometryReader { geometry in
            ZStack(alignment: .topLeading) {
                // Scrollable timeline content
                ScrollView([.horizontal, .vertical]) {
                    VStack(spacing: 0) {
                        // Time ruler
                        TimelineRulerView(
                            pixelsPerSecond: pixelsPerSecond,
                            duration: totalDuration
                        )
                        .frame(height: 30)

                        // Tracks
                        ForEach(editorState.clipTracks) { track in
                            HStack(spacing: 0) {
                                // Track header (fixed width)
                                ClipTrackHeaderView(track: track)
                                    .frame(width: 200, height: track.height)
                                    .background(Color.gray.opacity(0.2))

                                // Track content (scrollable)
                                ClipTrackContentView(
                                    track: track,
                                    pixelsPerSecond: pixelsPerSecond,
                                    selectedClipIDs: $selectedClipIDs
                                )
                                .frame(width: maxTimelineWidth, height: track.height)
                                .background(track.backgroundColor)
                                .onDrop(of: [.text], isTargeted: nil) { providers, location in
                                    handleDrop(on: track, at: location, providers: providers)
                                }
                            }
                        }
                    }
                }
                .background(Color.black)

                // Playhead overlay
                PlayheadView(
                    currentTime: editorState.currentTime,
                    pixelsPerSecond: pixelsPerSecond,
                    contentOffset: contentOffset
                )
                .frame(width: 1)
                .background(Color.red)

                // Snapping indicators
                if isSnapping {
                    SnappingOverlay(snapPoints: currentSnapPoints)
                }
            }
        }
    }

    private var totalDuration: CMTime {
        editorState.clipTracks
            .flatMap { $0.clips }
            .map { $0.timeRangeInTimeline.end }
            .max() ?? CMTime.zero
    }

    private var maxTimelineWidth: CGFloat {
        CGFloat(totalDuration.seconds) * pixelsPerSecond
    }
}
```

```swift
extension MultiClipTimelineView {
    private func dropPosition(from point: CGPoint, in track: ClipTrack) -> (trackID: UUID, time: CMTime) {
        // Calculate track index from Y coordinate
        let trackIndex = Int(point.y / track.height)
        let targetTrack = editorState.clipTracks[trackIndex]

        // Calculate timeline time from X coordinate
        let timeInSeconds = point.x / pixelsPerSecond
        let time = CMTime(
            seconds: Double(timeInSeconds),
            preferredTimescale: 600
        )

        return (targetTrack.id, time)
    }

    private func handleDrop(on track: ClipTrack, at location: CGPoint, providers: [NSItemProvider]) -> Bool {
        guard let provider = providers.first else { return false }

        provider.loadObject(ofClass: NSString.self) { object, _ in
            guard let clipIDString = object as? String,
                  let clipID = UUID(uuidString: clipIDString),
                  let clip = findClip(id: clipID) else { return }

            Task { @MainActor in
                let (trackID, time) = dropPosition(from: location, in: track)

                // Apply magnetic snapping
                let snappedTime = applyMagneticSnap(for: time, clipID: clipID)

                // Move clip to new position
                try? clipManager.moveClip(
                    clipID: clipID,
                    to: CMTimeRange(
                        start: snappedTime,
                        duration: clip.timelineDuration
                    ),
                    on: trackID,
                    ripple: false
                )
            }
        }

        return true
    }
}
```

**Magnetic Snapping Algorithm:**

```swift
extension MultiClipTimelineView {
    private let snapTolerance: CGFloat = 10.0  // 10 pixels

    private func applyMagneticSnap(for time: CMTime, clipID: UUID) -> CMTime {
        let snapPoints = calculateSnapPoints(excluding: clipID)
        let targetX = CGFloat(time.seconds) * pixelsPerSecond

        // Find closest snap point within tolerance
        for snapPoint in snapPoints {
            let snapX = CGFloat(snapPoint.seconds) * pixelsPerSecond
            let distance = abs(targetX - snapX)

            if distance <= snapTolerance {
                // Snap to this point
                return snapPoint
            }
        }

        // No snap point within tolerance
        return time
    }

    private func calculateSnapPoints(excluding clipID: UUID) -> [CMTime] {
        var points: [CMTime] = []

        // Snap to playhead
        points.append(editorState.currentTime)

        // Snap to chapter markers
        points.append(contentsOf: editorState.chapterMarkers.map { $0.time })

        // Snap to clip boundaries (start/end of all clips except this one)
        for track in editorState.clipTracks {
            for clip in track.clips where clip.id != clipID {
                points.append(clip.timeRangeInTimeline.start)
                points.append(clip.timeRangeInTimeline.end)
            }
        }

        return points.sorted()
    }
}
```
